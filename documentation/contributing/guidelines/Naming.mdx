  #engine
  #color

  // Create ‚úÖ
  create = ({ engine, color }) => {
    this.#engine = engine
    this.#color = color
    return this
  }

  // Read ‚úÖ
  read = () =>
    Object.assign(
      {},
      {
        engine: this.#engine,
        color: this.#color,
      }
    )

  // Update ‚úÖ

    engine && this.#engine == engine
    color && this.#color == color
    return this
  }

  // Delete ‚úÖ
  remove = () => {
    this.#engine = undefined
    this.#color = undefined
    return this
  }
}

### DataType directives

#### Booleans

Add **is** or **has** as variable name prefixes + participle verb or adjective:

```js
const name = true // ‚ùå
const isNamed = true // ‚úÖ

const red = true // ‚ùå
const isRed = true
```

Better use a semantic affirmative naming

```js
const isInvisible = false // ‚ùå
const isVisible = true // ‚úÖ

const hasRestrictions = false // ‚ùå
const hasPermissions = true // ‚úÖ
```

#### Number

Use singular
Use -**number** postfix on **n**- prefix

```js
const count = 0
const pageNumber = 3
const nPage = 3
```

#### Functions

Use a infinitive **verb**

```js

```

Is it async? Use gerund (-ing)

```js

```

Is it a React component prop handler? use the **on**-Whatever prefix

```js

```
